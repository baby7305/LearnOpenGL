cmake_minimum_required(VERSION 3.16.0)

project(learnopengl VERSION 1.0.0 LANGUAGES CXX C)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

find_package(Qt6 COMPONENTS Widgets REQUIRED)

add_subdirectory(packages)
include_directories(glad46/include)
include_directories(glm-master)

add_executable(hello_window
    glad46/src/glad.c
    main.cpp
)
add_executable(hello_triangle
        glad46/src/glad.c
        hello_triangle.cpp
)
add_executable(hello_triangle_exercise
        glad46/src/glad.c
        hello_triangle_exercise.cpp
)
add_executable(shaders_uniform
        glad46/src/glad.c
        shaders_uniform.cpp
)

target_link_libraries(hello_window glfw)
target_link_libraries(hello_triangle glfw)
target_link_libraries(hello_triangle_exercise glfw)
target_link_libraries(shaders_uniform glfw)
