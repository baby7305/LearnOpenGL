cmake_minimum_required(VERSION 3.16.0)

project(learnopengl VERSION 1.0.0 LANGUAGES CXX C)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

find_package(Qt6 COMPONENTS Widgets REQUIRED)
find_package(OpenCV REQUIRED)

add_subdirectory(packages)
include_directories(glad46/include)
include_directories(${OpenCV_INCLUDE_DIRS})

add_executable(hello_window
    glad46/src/glad.c
    main.cpp
)
add_executable(example_09-04
        glad46/src/glad.c
        example_09-04.cpp
)

target_link_libraries(hello_window glfw)
target_link_libraries(example_09-04 glfw ${OpenCV_LIBS})

SET(IMAGES ${CMAKE_SOURCE_DIR}/fruits.jpg)
FILE(COPY ${IMAGES} DESTINATION .)
FILE(COPY ${IMAGES} DESTINATION "Debug")
FILE(COPY ${IMAGES} DESTINATION "Release")
